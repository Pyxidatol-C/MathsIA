            \foreach \j in {0.5, 1.5, ..., 3.5} {
                \filldraw[black] (\i, \j) circle (1.5pt);
            }
        }
        
        \filldraw[gray] (3.0, 0.0) circle (1.5pt); 
        
        \node at (3.5, 0.25) {$(0, 0)$};
